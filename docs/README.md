from tkinter.filedialog import *
from tkinter import messagebox

# –ì–ª–æ–±–∞–ª—å–Ω—ã–µ –ø–µ—Ä–µ–º–µ–Ω–Ω—ã–µ

current_theme = "light"

# –§—É–Ω–∫—Ü–∏–∏
def newFile():
    global filename
    filename = "Untitled"
    text.delete(0.0, END)

def saveFile():
    global filename
    t = text.get(0.0, END)
    f = open(filename, 'w')
    f.write(t)
    f.close()

def saveAs():
    f = asksaveasfile(mode='w', defaultextension='.txt')
    t = text.get(0.0, END)
    try:
        f.write(t.rstrip())
    except:
        messagebox.showerror(title="Oops!", message="Unable to save file‚Ä¶")

def openFile():
    f = askopenfile(mode='r')
    t = f.read()
    text.delete(0.0, END)
    text.insert(0.0, t)

def setLightTheme():
    global current_theme
    text.config(bg="white", fg="black")
    root.config(bg="white")
    current_theme = "light"

def setDarkTheme():
    global current_theme
    text.config(bg="#2e2e2e", fg="white")
    root.config(bg="#2e2e2e")
    current_theme = "dark"

def copyText():
    text.clipboard_clear()
    text.clipboard_append(text.get(0.0, END))

def pasteText():
    try:
        clipboard_text = root.clipboard_get()
        text.insert(INSERT, clipboard_text)
    except:
        messagebox.showerror("–û—à–∏–±–∫–∞", "–ù–µ —É–¥–∞–µ—Ç—Å—è –≤—Å—Ç–∞–≤–∏—Ç—å —Ç–µ–∫—Å—Ç")

def cutText():
    copyText()
    text.delete(0.0, END)

# –°–æ–∑–¥–∞–Ω–∏–µ –æ–∫–Ω–∞
root = Tk()
root.title("My Python Text Editor")
root.minsize(width=400, height=400)
root.maxsize(width=400, height=400)

# –¢–µ–∫—Å—Ç–æ–≤–æ–µ –ø–æ–ª–µ
text = Text(root, width=400, height=400)
text.pack()

# –ú–µ–Ω—é
menubar = Menu(root)

filemenu = Menu(menubar)
filemenu.add_command(label="–ù–æ–≤—ã–π", command=newFile)
filemenu.add_command(label="–û—Ç–∫—Ä—ã—Ç—å", command=openFile)
filemenu.add_command(label="–°–æ—Ö—Ä–∞–Ω–∏—Ç—å", command=saveFile)
filemenu.add_command(label="–°–æ—Ö—Ä–∞–Ω–∏—Ç—å –∫–∞–∫‚Ä¶", command=saveAs)
filemenu.add_separator()
filemenu.add_command(label="–í—ã—Ö–æ–¥", command=root.quit)
menubar.add_cascade(label="–§–∞–π–ª", menu=filemenu)

editmenu = Menu(menubar)
editmenu.add_command(label="–ö–æ–ø–∏—Ä–æ–≤–∞—Ç—å", command=copyText)
editmenu.add_command(label="–í—Å—Ç–∞–≤–∏—Ç—å", command=pasteText)
editmenu.add_command(label="–í—ã—Ä–µ–∑–∞—Ç—å", command=cutText)
menubar.add_cascade(label="–†–µ–¥–∞–∫—Ç–∏—Ä–æ–≤–∞—Ç—å", menu=editmenu)

viewmenu = Menu(menubar)
viewmenu.add_command(label="–°–≤–µ—Ç–ª–∞—è", command=setLightTheme)
viewmenu.add_command(label="–¢–µ–º–Ω–∞—è", command=setDarkTheme)
menubar.add_cascade(label="–¢–µ–º–∞", menu=viewmenu)

root.config(menu=menubar)
root.mainloop()
